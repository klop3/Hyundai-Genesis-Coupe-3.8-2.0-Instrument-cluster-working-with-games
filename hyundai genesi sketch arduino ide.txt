#ifndef SHCustomProtocol_h
#define SHCustomProtocol_h

#include <SPI.h>
#include <mcp_can.h>

static const int CAN_SPI_CS_PIN = 10;
MCP_CAN SH_CAN(CAN_SPI_CS_PIN);

class SHCustomProtocol {
private:

  // ─── Gear data ───────────────────────────────────────────────────────────
  byte data1[8]       = {0x19,0x38,0x65,0x62,0xA3,0xED,0x86,0};
  byte data2[8]       = {0xAA,0x38,0x6B,0x9E,0x6E,0xF7,0x76,0};
  byte data3[8]       = {0x23,0xE8,0x8A,0x76,0x39,0x61,0x9B,0};
  byte data4[8]       = {0xEC,0xE8,0x24,0x06,0xBE,0x27,0x7C,0};
  byte data5[8]       = {0xBD,0x78,0xE6,0xA0,0x5F,0x55,0x9C,0};
  byte data6[8]       = {0xCE,0xD8,0xCF,0xF6,0x9D,0xBC,0x23,0};
  byte dataNeutral[8] = {0x70,0xE6,0xE2,0x98,0x97,0xBE,0x7E,0xDE};
  byte dataR[8]       = {0xC5,0x57,0x9F,0x6D,0xFF,0xEF,0xE0,0x01};
  byte* gearData[8]   = {data1,data2,data3,data4,data5,data6,dataNeutral,dataR};
  byte gearCounters[8]= {0,0,0,0,0,0,0,0};

  // ─── CAN frame buffers ───────────────────────────────────────────────────
  byte dataRPM[8]  = {0,0,0,0,0,0,0,0};
  byte data1F1[8]  = {0,0,0,0,0,0,0,0};
  byte data329[8]  = {0,0,0,0,0,0,0,0};
  byte zero[8]     = {0,0,0,0,0,0,0,0};

  // ─── State flags ─────────────────────────────────────────────────────────
  unsigned long lastDataTime    = 0;
  bool startSequenceDone        = false;
  bool sweepGuard               = false;
  bool canReady                 = false;

  int  lastValidRPM             = 0;
  int  lastValidSpeed           = 0;
  int  lastValidWater           = 0;

  bool absActive                = false;
  bool parkingBrake             = false;
  bool engineRunning            = false;
  bool engineStartedPrev        = false;

  bool turnLeft                 = false;
  bool turnRight                = false;
  bool highBeam                 = false;
  bool tyrePuncture             = false;
  bool lastTyrePuncture         = false;

  bool tcActive                 = false;
  bool lastTcActive             = false;
  unsigned long tcEndTime       = 0;
  bool tcJustEnded              = false;

  // ─── TC systém ON/OFF (klíč / Drift mód) ─────────────────────────────────
  static const unsigned long TC_ACTIVE_TIMEOUT = 5000;
  unsigned long lastTcActiveTime = 0;

  char lastValidGear            = 'N';

  float odometer                = 0.0f;
  unsigned long lastOdoTime     = 0;
  float smoothWater             = 0.0f;

  // Anti-glitch: confirmed zero speed after 2 consecutive zero frames
  int zeroSpeedCount            = 0;

  // Tyre puncture alert timing
  unsigned long tyreAlertStart  = 0;
  bool tyreAlertActive          = false;

  // ─── Helpers ─────────────────────────────────────────────────────────────

  byte waterTempToByte(float tempC) {
    float val = (tempC + 48.373f) / 0.75f;
    return (byte)constrain((int)val, 0, 255);
  }

  int mapGearChar(char g) {
    switch (g) {
      case '1': return 0;
      case '2': return 1;
      case '3': return 2;
      case '4': return 3;
      case '5': return 4;
      case '6': return 5;
      case 'N':
      case 'P': return 6;
      case 'R': return 7;
      default:  return 6;
    }
  }

  // ─── 0x1F1 + 0x316 ───────────────────────────────────────────────────────

  void sendRealValues() {
    unsigned long odoM = (unsigned long)odometer;

    byte b0;
    unsigned long now = millis();

    if (tcActive) {
      b0 = ((now / 250) % 2 == 0) ? 0x50 : 0x20;
      tcJustEnded = false;
    } else if (tcJustEnded) {
      bool tcWasRecentlyActive = (lastTcActiveTime > 0) &&
                                 (now - lastTcActiveTime < TC_ACTIVE_TIMEOUT);

      if (tcWasRecentlyActive && (now - tcEndTime < 1000)) {
        b0 = 0x00;
      } else {
        b0 = 0x00;
        tcJustEnded = false;
      }
    } else {
      b0 = 0x20;
      if (absActive)    b0 |= 0x01;
      if (parkingBrake) b0 |= 0x02;
    }

    data1F1[0] = b0;
    data1F1[1] = (byte)(odoM & 0xFF);
    data1F1[2] = (byte)((odoM >> 8)  & 0xFF);
    data1F1[3] = (byte)((odoM >> 16) & 0xFF);

    if (lastValidSpeed <= 0) {
      zeroSpeedCount++;
      if (zeroSpeedCount >= 2) {
        data1F1[4] = 0x00;
      }
    } else {
      zeroSpeedCount = 0;
      // Opravený speed výpočet: 0 km/h = 0x00, 20 km/h = 0x13, 40 km/h = 0x25
      // Lineární: byte = km/h * 0.95
      data1F1[4] = (byte)constrain((int)(lastValidSpeed * 0.95f), 0, 255);
    }

    data1F1[5] = 0x20;
    SH_CAN.sendMsgBuf(0x1F1, 0, 8, data1F1);

    unsigned int rpmVal = (unsigned int)(lastValidRPM / 62.5f);
    dataRPM[2] = (rpmVal >> 8) & 0xFF;
    dataRPM[3] =  rpmVal       & 0xFF;
    SH_CAN.sendMsgBuf(0x316, 0, 8, dataRPM);
  }

  // ─── 0x5F0 — tyre puncture: 3s blink → steady ───────────────────────────

  void sendTyreState() {
    byte b0;

    if (!tyrePuncture) {
      b0 = 0x00;
      tyreAlertActive  = false;
      tyreAlertStart   = 0;
      lastTyrePuncture = false;
    } else {
      if (!lastTyrePuncture) {
        tyreAlertActive  = true;
        tyreAlertStart   = millis();
        lastTyrePuncture = true;
      }

      unsigned long elapsed = millis() - tyreAlertStart;

      if (elapsed < 3000) {
        b0 = ((elapsed / 500) % 2 == 0) ? 0x18 : 0x00;
      } else {
        b0 = 0x18;
      }
    }

    byte msg5F0[8] = {b0, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00};
    SH_CAN.sendMsgBuf(0x5F0, 0, 8, msg5F0);
  }

  // ─── Needle sweep on engine start ────────────────────────────────────────

  void needleSweep() {
    const unsigned int RPM_MAX_VAL = (unsigned int)(8000.0f / 62.5f);

    lastDataTime = millis();

    for (int b = 0; b <= 255; b += 3) {
      unsigned int rSweep = (unsigned int)((b / 255.0f) * RPM_MAX_VAL);
      byte tmpRPM[8] = {0, 0, (byte)(rSweep >> 8), (byte)(rSweep & 0xFF), 0, 0, 0, 0};
      byte tmpSPD[8] = {0x00, 0, 0, 0, (byte)b, 0xFF, 0, 0};
      SH_CAN.sendMsgBuf(0x316, 0, 8, tmpRPM);
      SH_CAN.sendMsgBuf(0x1F1, 0, 8, tmpSPD);
      delay(10);
    }

    {
      byte tmpRPM[8] = {0, 0, (byte)(RPM_MAX_VAL >> 8), (byte)(RPM_MAX_VAL & 0xFF), 0, 0, 0, 0};
      byte tmpSPD[8] = {0x00, 0, 0, 0, 0xFF, 0xFF, 0, 0};
      SH_CAN.sendMsgBuf(0x316, 0, 8, tmpRPM);
      SH_CAN.sendMsgBuf(0x1F1, 0, 8, tmpSPD);
    }

    lastDataTime = millis();
    delay(300);

    for (int b = 255; b >= 0; b -= 4) {
      unsigned int rSweep = (unsigned int)((b / 255.0f) * RPM_MAX_VAL);
      byte tmpRPM[8] = {0, 0, (byte)(rSweep >> 8), (byte)(rSweep & 0xFF), 0, 0, 0, 0};
      byte tmpSPD[8] = {0x00, 0, 0, 0, (byte)b, 0xFF, 0, 0};
      SH_CAN.sendMsgBuf(0x316, 0, 8, tmpRPM);
      SH_CAN.sendMsgBuf(0x1F1, 0, 8, tmpSPD);
      delay(10);
    }

    byte tmpZeroRPM[8] = {0, 0, 0, 0, 0, 0, 0, 0};
    byte tmpZeroSPD[8] = {0x1F, 0, 0, 0, 0x00, 0x20, 0, 0};
    SH_CAN.sendMsgBuf(0x316, 0, 8, tmpZeroRPM);
    SH_CAN.sendMsgBuf(0x1F1, 0, 8, tmpZeroSPD);
  }

  // ─── Off state (engine off / timeout) ────────────────────────────────────

  void sendOffState() {
    SH_CAN.sendMsgBuf(0x316, 0, 8, zero);

    byte speedZero[8] = {0x1F, 0, 0, 0, 0x00, 0x20, 0, 0};
    SH_CAN.sendMsgBuf(0x1F1, 0, 8, speedZero);

    byte gearOff[8] = {0xFF,0xFF,0xFF,0xFF,0xFF,0xFF,0xFF,0xFF};
    SH_CAN.sendMsgBuf(0x43F, 0, 8, gearOff);

    byte msg5F0off[8] = {0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00};
    SH_CAN.sendMsgBuf(0x5F0, 0, 8, msg5F0off);
  }

public:

  // ─── setup ───────────────────────────────────────────────────────────────

  void setup() {
    while (SH_CAN.begin(MCP_ANY, CAN_500KBPS, MCP_8MHZ) != CAN_OK) {
      delay(100);
    }
    SH_CAN.setMode(MCP_NORMAL);
    lastOdoTime = millis();
    canReady = true;
  }

  // ─── read ────────────────────────────────────────────────────────────────

  void read() {
    if (!canReady) return;
    String msg = FlowSerialReadStringUntil('\n');
    msg.trim();
    if (msg.length() == 0) return;
    parseAndProcess(msg);
  }

  // ─── parseAndProcess ─────────────────────────────────────────────────────
  //
  // SimHub formule (14 polí, 13 středníků):
  //
  //  [0]  Gear
  //  [1]  Rpms
  //  [2]  SpeedKmh
  //  [3]  WaterTemperature
  //  [4]  0  (reserved)
  //  [5]  AbsActive
  //  [6]  EngineRunning
  //  [7]  0  (reserved)
  //  [8]  TurnIndicatorLeft
  //  [9]  TurnIndicatorRight
  //  [10] HighBeam
  //  [11] TyrePuncture
  //  [12] Handbrake
  //  [13] TCActive
  //

  void parseAndProcess(String& msg) {
    int s1  = msg.indexOf(';');
    int s2  = msg.indexOf(';', s1  + 1);
    int s3  = msg.indexOf(';', s2  + 1);
    int s4  = msg.indexOf(';', s3  + 1);
    int s5  = msg.indexOf(';', s4  + 1);
    int s6  = msg.indexOf(';', s5  + 1);
    int s7  = msg.indexOf(';', s6  + 1);
    int s8  = msg.indexOf(';', s7  + 1);
    int s9  = msg.indexOf(';', s8  + 1);
    int s10 = msg.indexOf(';', s9  + 1);
    int s11 = msg.indexOf(';', s10 + 1);
    int s12 = msg.indexOf(';', s11 + 1);
    int s13 = msg.indexOf(';', s12 + 1);

    if (s1 < 0 || s13 < 0) return;

    lastDataTime   = millis();
    lastValidGear  = msg.charAt(0);
    lastValidRPM   = msg.substring(s1  + 1, s2).toInt();
    lastValidSpeed = msg.substring(s2  + 1, s3).toInt();
    lastValidWater = msg.substring(s3  + 1, s4).toInt();
    absActive      = msg.substring(s5  + 1, s6).toInt() != 0;
    engineRunning  = msg.substring(s6  + 1, s7).toInt() != 0;
    turnLeft       = msg.substring(s8  + 1, s9).toInt()  != 0;
    turnRight      = msg.substring(s9  + 1, s10).toInt() != 0;
    highBeam       = msg.substring(s10 + 1, s11).toInt() != 0;
    tyrePuncture   = msg.substring(s11 + 1, s12).toInt() != 0;
    parkingBrake   = msg.substring(s12 + 1, s13).toInt() != 0;

    bool newTc = msg.substring(s13 + 1).toInt() != 0;

    if (newTc) {
      lastTcActiveTime = millis();
    }

    if (lastTcActive && !newTc) {
      tcJustEnded = true;
      tcEndTime   = millis();
    }

    lastTcActive = newTc;
    tcActive     = newTc;

    if (engineRunning && !engineStartedPrev && !startSequenceDone && lastValidRPM >= 1) {
      needleSweep();
      startSequenceDone = true;
      sweepGuard = true;
    }

    if (sweepGuard && lastValidRPM == 0 && !engineRunning) {
      startSequenceDone = false;
      sweepGuard = false;
    }

    engineStartedPrev = engineRunning;
  }

  // ─── loop ────────────────────────────────────────────────────────────────

  void loop() {
    if (!canReady) return;

    bool timeout = (millis() - lastDataTime > 2000);

    if (!engineRunning || timeout) {
      zeroSpeedCount   = 0;
      tyreAlertActive  = false;
      tyreAlertStart   = 0;
      tyrePuncture     = false;
      lastTyrePuncture = false;
      tcActive         = false;
      lastTcActive     = false;
      tcJustEnded      = false;
      lastTcActiveTime = 0;
      sendOffState();
    } else {
      unsigned long now = millis();
      float dt = (now - lastOdoTime) / 1000.0f;
      lastOdoTime = now;
      if (lastValidSpeed > 0) odometer += (lastValidSpeed / 3.6f) * dt;

      float rate = (lastValidWater > smoothWater) ? 0.005f : 0.02f;
      smoothWater += (lastValidWater - smoothWater) * rate;
      data329[1] = waterTempToByte(smoothWater);
      SH_CAN.sendMsgBuf(0x329, 0, 8, data329);

      int gIdx = mapGearChar(lastValidGear);
      gearData[gIdx][7] = gearCounters[gIdx]++;
      SH_CAN.sendMsgBuf(0x43F, 0, 8, gearData[gIdx]);

      sendRealValues();
      sendTyreState();
    }

    delay(1);
  }

  void idle() { }
};

#endif
